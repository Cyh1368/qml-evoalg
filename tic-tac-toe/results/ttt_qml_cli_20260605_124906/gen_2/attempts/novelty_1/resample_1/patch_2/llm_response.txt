<DESCRIPTION>
Add a parameter-shared CRX ring entanglement on the outer perimeter after the CZ ring, increasing expressivity without increasing parameter count. All eight outer-edge CRX gates share a single parameter ring_crx. This preserves the EVOLVE-BLOCK constraints (fixed data map, re-uploading, hub CRZ, post-RZ, etc.) while potentially improving validation performance and generalization by enriching two-qubit interactions around the board perimeter.

def _build_hierarchical_ansatz_spec() -> list[dict]:
    """Build hierarchical entanglement ansatz with parameter sharing.
    
    Structure:
    1. Pre-entanglement RZ on all qubits (shared by symmetry)
    2. RY rotations on all qubits (shared by symmetry)
    3. Ring entanglement: CZ gates around outer ring
    4. Hub entanglement: CRZ gates from center to edges
    5. Post-entanglement RZ rotations on all qubits (shared by symmetry)
    """
    spec = []
    
    # Symmetry groups for 4-fold rotational symmetry of tic-tac-toe board
    # Corners: 0, 2, 4, 6 (all equivalent by rotation)
    # Edges: 1, 3, 5, 7 (all equivalent by rotation)
    # Center: 8 (unique)
    
    # Layer 1: Pre-entanglement RZ rotations with parameter sharing
    spec.append({"gate": "RZ", "wire": 0, "param": "rz_pre_corner"})
    spec.append({"gate": "RZ", "wire": 2, "param": "rz_pre_corner"})
    spec.append({"gate": "RZ", "wire": 4, "param": "rz_pre_corner"})
    spec.append({"gate": "RZ", "wire": 6, "param": "rz_pre_corner"})
    spec.append({"gate": "RZ", "wire": 1, "param": "rz_pre_edge"})
    spec.append({"gate": "RZ", "wire": 3, "param": "rz_pre_edge"})
    spec.append({"gate": "RZ", "wire": 5, "param": "rz_pre_edge"})
    spec.append({"gate": "RZ", "wire": 7, "param": "rz_pre_edge"})
    spec.append({"gate": "RZ", "wire": 8, "param": "rz_pre_center"})
    
    # Layer 2: RY rotations with parameter sharing
    spec.append({"gate": "RY", "wire": 0, "param": "ry_corner"})
    spec.append({"gate": "RY", "wire": 2, "param": "ry_corner"})
    spec.append({"gate": "RY", "wire": 4, "param": "ry_corner"})
    spec.append({"gate": "RY", "wire": 6, "param": "ry_corner"})
    spec.append({"gate": "RY", "wire": 1, "param": "ry_edge"})
    spec.append({"gate": "RY", "wire": 3, "param": "ry_edge"})
    spec.append({"gate": "RY", "wire": 5, "param": "ry_edge"})
    spec.append({"gate": "RY", "wire": 7, "param": "ry_edge"})
    spec.append({"gate": "RY", "wire": 8, "param": "ry_center"})
    
    # Layer 3: Ring entanglement (outer 8 qubits)
    # CZ gates around the perimeter
    spec.append({"gate": "CZ", "wires": [0, 1]})
    spec.append({"gate": "CZ", "wires": [1, 2]})
    spec.append({"gate": "CZ", "wires": [2, 3]})
    spec.append({"gate": "CZ", "wires": [3, 4]})
    spec.append({"gate": "CZ", "wires": [4, 5]})
    spec.append({"gate": "CZ", "wires": [5, 6]})
    spec.append({"gate": "CZ", "wires": [6, 7]})
    spec.append({"gate": "CZ", "wires": [7, 0]})
    
    # Layer 4: Hub entanglement (center to edges with parametrized gates)
    # Use CRZ for adaptive coupling between center and edges
    spec.append({"gate": "CRZ", "wires": [8, 1], "param": "crz_hub_edge"})
    spec.append({"gate": "CRZ", "wires": [8, 3], "param": "crz_hub_edge"})
    spec.append({"gate": "CRZ", "wires": [8, 5], "param": "crz_hub_edge"})
    spec.append({"gate": "CRZ", "wires": [8, 7], "param": "crz_hub_edge"})
    
    # Layer 5: Post-entanglement RZ rotations with parameter sharing
    spec.append({"gate": "RZ", "wire": 0, "param": "rz_post_corner"})
    spec.append({"gate": "RZ", "wire": 2, "param": "rz_post_corner"})
    spec.append({"gate": "RZ", "wire": 4, "param": "rz_post_corner"})
    spec.append({"gate": "RZ", "wire": 6, "param": "rz_post_corner"})
    spec.append({"gate": "RZ", "wire": 1, "param": "rz_post_edge"})
    spec.append({"gate": "RZ", "wire": 3, "param": "rz_post_edge"})
    spec.append({"gate": "RZ", "wire": 5, "param": "rz_post_edge"})
    spec.append({"gate": "RZ", "wire": 7, "param": "rz_post_edge"})
    spec.append({"gate": "RZ", "wire": 8, "param": "rz_post_center"})
    
    return spec
