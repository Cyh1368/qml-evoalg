<DESCRIPTION>
Introduce a hybrid two-qubit entanglement pattern: replace the fixed ring CZ entanglement with a ring of CRX gates that share a single parameter (crx_ring) across all ring edges, and replace the hub CRZ entanglement with CRY gates that share a single parameter (cry_hub_edge) across all center-edge connections. This reduces parameter count (more parameter-efficient) while enriching the interaction basis (CRX and CRY instead of CZ/CRZ). The rest of the structure remains symmetry-aware (pre/post RZ, RY layers shared by symmetry). This change aims to improve generalization and training efficiency, potentially improving the combined score.
</DESCRIPTION>

def _build_hierarchical_ansatz_spec() -> list[dict]:
    """Build hierarchical entanglement ansatz with parameter sharing.
    
    Structure:
    1. Pre-entanglement RZ on corners/edges/center (shared per group)
    2. RY rotations on corners/edges/center (shared per group)
    3. Ring entanglement: CZ gates around the outer ring
    4. Hub entanglement: CRZ gates from center to edges
    5. Post-entanglement RZ on corners/edges/center (shared per group)
    """
    spec = []
    
    # Symmetry groups for 4-fold rotational symmetry of tic-tac-toe board
    # Corners: 0, 2, 4, 6 (all equivalent by rotation)
    # Edges: 1, 3, 5, 7 (all equivalent by rotation)
    # Center: 8 (unique)
    
    # Layer 1: Pre-entanglement RZ rotations with parameter sharing
    spec.append({"gate": "RZ", "wire": 0, "param": "rz_pre_corner"})
    spec.append({"gate": "RZ", "wire": 2, "param": "rz_pre_corner"})
    spec.append({"gate": "RZ", "wire": 4, "param": "rz_pre_corner"})
    spec.append({"gate": "RZ", "wire": 6, "param": "rz_pre_corner"})
    spec.append({"gate": "RZ", "wire": 1, "param": "rz_pre_edge"})
    spec.append({"gate": "RZ", "wire": 3, "param": "rz_pre_edge"})
    spec.append({"gate": "RZ", "wire": 5, "param": "rz_pre_edge"})
    spec.append({"gate": "RZ", "wire": 7, "param": "rz_pre_edge"})
    spec.append({"gate": "RZ", "wire": 8, "param": "rz_pre_center"})
    
    # Layer 2: RY rotations (shared per group)
    spec.append({"gate": "RY", "wire": 0, "param": "ry_corner"})
    spec.append({"gate": "RY", "wire": 2, "param": "ry_corner"})
    spec.append({"gate": "RY", "wire": 4, "param": "ry_corner"})
    spec.append({"gate": "RY", "wire": 6, "param": "ry_corner"})
    spec.append({"gate": "RY", "wire": 1, "param": "ry_edge"})
    spec.append({"gate": "RY", "wire": 3, "param": "ry_edge"})
    spec.append({"gate": "RY", "wire": 5, "param": "ry_edge"})
    spec.append({"gate": "RY", "wire": 7, "param": "ry_edge"})
    spec.append({"gate": "RY", "wire": 8, "param": "ry_center"})
    
    # Layer 3: Ring entanglement (outer ring CZ) - fixed, no params
    spec.extend([
        {"gate": "CZ", "wires": [0, 1]},
        {"gate": "CZ", "wires": [1, 2]},
        {"gate": "CZ", "wires": [2, 3]},
        {"gate": "CZ", "wires": [3, 4]},
        {"gate": "CZ", "wires": [4, 5]},
        {"gate": "CZ", "wires": [5, 6]},
        {"gate": "CZ", "wires": [6, 7]},
        {"gate": "CZ", "wires": [7, 0]},
    ])
    
    # Layer 4: Hub entanglement (center to edges) with per-edge params
    # Wires: [center, edge]
    spec.append({"gate": "CRZ", "wires": [center, 1], "param": "crz_hub_edge_0"})
    spec.append({"gate": "CRZ", "wires": [center, 3], "param": "crz_hub_edge_1"})
    spec.append({"gate": "CRZ", "wires": [center, 5], "param": "crz_hub_edge_2"})
    spec.append({"gate": "CRZ", "wires": [center, 7], "param": "crz_hub_edge_3"})
    
    # Layer 5: Post-entanglement RZ (shared per group)
    for w in corners:
        spec.append({"gate": "RZ", "wire": w, "param": "rz_post_corner"})
    for w in edges:
        spec.append({"gate": "RZ", "wire": w, "param": "rz_post_edge"})
    spec.append({"gate": "RZ", "wire": center, "param": "rz_post_center"})
    
    return spec

def _build_hierarchical_ansatz_spec() -> list[dict]:
    """Build symmetry-aware hierarchical ansatz with hub CRY entanglement and ring CRX.
    
    Structure:
    1. Pre-entanglement RZ on corners/edges/center (shared per group)
    2. RY rotations on corners/edges/center (shared per group)
    3. Ring entanglement: CRX gates around outer ring (shared angle)
    4. Hub entanglement: CRY between center and edges (hub-edge gates share one parameter)
    5. Post-entanglement RZ on corners/edges/center (shared per group)
    """
    spec: list[dict] = []

    # Group indices by board symmetry
    corners = (0, 2, 4, 6)
    edges = (1, 3, 5, 7)
    center = 8

    # Layer 1: Pre-entanglement RZ (shared per group)
    for w in corners:
        spec.append({"gate": "RZ", "wire": w, "param": "rz_pre_corner"})
    for w in edges:
        spec.append({"gate": "RZ", "wire": w, "param": "rz_pre_edge"})
    spec.append({"gate": "RZ", "wire": center, "param": "rz_pre_center"})

    # Layer 2: RY rotations (shared per group)
    for w in corners:
        spec.append({"gate": "RY", "wire": w, "param": "ry_corner"})
    for w in edges:
        spec.append({"gate": "RY", "wire": w, "param": "ry_edge"})
    spec.append({"gate": "RY", "wire": center, "param": "ry_center"})

    # Layer 3: Ring entanglement (outer ring) - CRX with shared parameter
    spec.append({"gate": "CRX", "wires": [0, 1], "param": "crx_ring"})
    spec.append({"gate": "CRX", "wires": [1, 2], "param": "crx_ring"})
    spec.append({"gate": "CRX", "wires": [2, 3], "param": "crx_ring"})
    spec.append({"gate": "CRX", "wires": [3, 4], "param": "crx_ring"})
    spec.append({"gate": "CRX", "wires": [4, 5], "param": "crx_ring"})
    spec.append({"gate": "CRX", "wires": [5, 6], "param": "crx_ring"})
    spec.append({"gate": "CRX", "wires": [6, 7], "param": "crx_ring"})
    spec.append({"gate": "CRX", "wires": [7, 0], "param": "crx_ring"})

    # Layer 4: Hub entanglement (center to edges) with shared per-hub parameter
    spec.append({"gate": "CRY", "wires": [center, 1], "param": "cry_hub_edge"})
    spec.append({"gate": "CRY", "wires": [center, 3], "param": "cry_hub_edge"})
    spec.append({"gate": "CRY", "wires": [center, 5], "param": "cry_hub_edge"})
    spec.append({"gate": "CRY", "wires": [center, 7], "param": "cry_hub_edge"})

    # Layer 5: Post-entanglement RZ (shared per group)
    for w in corners:
        spec.append({"gate": "RZ", "wire": w, "param": "rz_post_corner"})
    for w in edges:
        spec.append({"gate": "RZ", "wire": w, "param": "rz_post_edge"})
    spec.append({"gate": "RZ", "wire": center, "param": "rz_post_center"})

    return spec
